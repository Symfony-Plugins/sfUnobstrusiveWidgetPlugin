# learn by example - double list


## create widget

$doubleList = new sfUoWidgetFormSelectMany(
  array(
    'choices' => array(1 => 'item 1', 2 => 'item 2', 3 => 'item 3', 4 => 'item 4', 5 => 'item 5', 6 => 'item 6', 7 => 'item 7', 8 => 'item 8', 9 => 'item 9', 10 => 'item 10'), 
    'js_transformer'=>'double_list'
  )
);


## render widget

$doubleList->render('form_field_name');
